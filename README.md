### ⚡ Quick Start (30 seconds)

```bash
# 1. Clone the repo
git clone https://github.com/yourusername/bug-bounty-dorks.git
cd bug-bounty-dorks

# 2. Pick your level
cat dorks.txt | grep "Level 1"   # Beginner

```

### 🔧 Advanced Usage

```bash
# Automation with bash
cat dorks_clean.txt | while read dork; do
    echo "[+] Searching: $dork"
    # Add your scraping logic here
done

# Fuzzing bug bounty endpoints
gobuster dir -u https://target.com   -w bug-bounty-paths.txt   -t 50

# Subdomain enumeration for security pages
subfinder -d target.com -all |   httpx -path "/security.txt,/bug-bounty,/responsible-disclosure" -sc

# Mass scanning with nuclei

```
# Exclude major platforms (find self-hosted)

# Filter by reward type

# Filter by region + reward

# Find recently launched programs

# Cache analysis for historical data

### 🔍 Subdomain Enumeration

```bash
# Common bug bounty subdomains to enumerate
subfinder -d target.com -all | tee subdomains.txt

# Then check these patterns:
# security.target.com
# bugbounty.target.com
# responsible-disclosure.target.com
# security-researchers.target.com
# report-vulnerability.target.com
```

### 🤖 Full Automation Pipeline

```bash
#!/bin/bash
# auto-bounty-recon.sh

TARGET=$1

echo "[+] Starting recon for: $TARGET"

# Subdomain enumeration
echo "[*] Enumerating subdomains..."
subfinder -d $TARGET -all -silent | anew subs.txt

# Probe for live hosts
echo "[*] Probing live hosts..."
cat subs.txt | httpx -silent | anew live.txt

# Check for security pages
echo "[*] Checking security pages..."
cat live.txt | while read url; do
    curl -s "$url/security.txt" | grep -iE "contact|bounty|reward" && echo "[+] Found: $url/security.txt"
    curl -s "$url/.well-known/security.txt" | grep -iE "contact|bounty|reward" && echo "[+] Found: $url/.well-known/security.txt"
    curl -s "$url/bug-bounty" -I | grep "200" && echo "[+] Found: $url/bug-bounty"
    curl -s "$url/responsible-disclosure" -I | grep "200" && echo "[+] Found: $url/responsible-disclosure"
done

echo "[+] Recon complete!"
```
